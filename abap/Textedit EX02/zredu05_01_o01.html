MODULE active_tab OUTPUT.
  go_tab-activetab = gv_pressedtab.

  CASE gv_pressedtab.
    WHEN 'T01'.
      gv_subscreen = '0101'.
    WHEN 'T02'.
      gv_subscreen = '0102'.
  ENDCASE.

ENDMODULE.
