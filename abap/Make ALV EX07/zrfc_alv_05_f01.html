FORM get_data .

  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE gt_data
    FROM zppg0150t
    WHERE werks EQ p_werks
      AND budat IN s_budat.

  IF gt_data[] IS INITIAL.
    MESSAGE s000 WITH 'Data not found'.
    LEAVE LIST-PROCESSING.
  ENDIF.
ENDFORM.
